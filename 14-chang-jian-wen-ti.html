public static BO RsetToTable(ResultSet rs) {
        BO bo1 = new BO();
        try {
            ResultSetMetaData rsmd = rs.getMetaData();
            // 得到当前的列数
            int colCount = rsmd.getColumnCount();
            for (int i = 1; i <= colCount; i++) {// for循环控制列数
                String name = rsmd.getColumnName(i).toUpperCase();// 得到当前列的值
                String value = rs.getString(i);// 得到当前列的列名
                if (value == null || value.length() < 1) {
                    bo1.set(name, "");
                } else {
                    bo1.set(name, value);
                }
            }

        } catch (Exception e) {
            e.printStackTrace();
        }
        return bo1;
    }
}

                     callback:弹出消息提示后回调函数。

             } );
